/**
  * Faz um request HTTP de acordo com o método de entrada para a URL. Em seguida, executa o callback de entrada.
  * @param {string} url - A URL onde o método será chamado.
  * @param {string} method - O verbo HTTP (POST, GET, etc).
  * @param {string} callback - O callback que será chamado após o método.
  */
function ajax(url, method, callback) {
  let request = new XMLHttpRequest();
  request.overrideMimeType("application/json");
  request.open(method, url, true);
  request.onreadystatechange = () => {
    if (request.readyState === 4 && request.status == "200") {
      callback(request.responseText);
    }
  };
  request.send(null);
}
